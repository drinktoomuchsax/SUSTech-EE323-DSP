A. Case 1: Electric field distribution of two identical point charges;

1. Matlab source code. 
%% CASE1

% Constants and parameters
k = 8.99e9;  % Electrostatic constant
Q1 = 1e-9;   % Charge 1 (in Coulombs)
Q2 = 1e-9;   % Charge 2 (in Coulombs)
x_range = linspace(-0.02, 0.02, 100);  % Field range
y_range = linspace(-0.02, 0.02, 100);

% Calculate electric potential
[X, Y] = meshgrid(x_range, y_range);
r1 = sqrt((X + 0.01).^2 + Y.^2);
r2 = sqrt((X - 0.01).^2 + Y.^2);
V = k * (Q1./r1 + Q2./r2);



% Plot the distribution of electric potential
figure;
mesh(X, Y, V);
hold on;
title('Potential distribution of each point - Case 1 - 12110623 曹正阳', 'FontSize', 10);
xlabel('X-axis (m)', 'FontSize', 20);
ylabel('Y-axis (m)', 'FontSize', 20);



% Define additional contour levels
contour_levels = 300;  % Adjust as needed

% Plot the electric potential distribution
figure;
contour(X, Y, V, contour_levels, 'LineWidth', 1.5);
title('Distribution of the contours - Case 1 - 12110623 曹正阳');
xlabel('X-axis (m)');
ylabel('Y-axis (m)');
colorbar;



% Calculate electric field
[Ex, Ey] = gradient(-V);



% Create a figure for streamlines and equipotential lines
figure;
grid on;
% Define the angle difference between adjacent field lines
del_theta = 7;

% Create angles in radians
theta = (0:del_theta:360) .* pi / 180;

% Generate x and y coordinates for the start of the field lines
xs = 0.0001 * cos(theta);
ys = 0.0001 * sin(theta);
streamline(X, Y, Ex, Ey, [xs+0.01 xs-0.01] ,[ys ys] );
hold on;
contour(X, Y, V, 1);  % Veq should be defined based on your specific case
plot(0, 0, 'o', 'MarkerSize', 12);  % Plot the point charge at the origin



% Set title and labels
title('Distribution of equipotential lines and steamlines with smooth continuous curves - Case 1 - 12110623 曹正阳');
xlabel('X-axis (m)', 'FontSize', 12);
ylabel('Y-axis (m)', 'FontSize', 12);

% Normalize the electric field vectors
E_mag = sqrt(Ex.^2 + Ey.^2);
Ex_norm = Ex ./ E_mag;
Ey_norm = Ey ./ E_mag;

% Subsample the grid for arrows
arrow_spacing = 4;  % Adjust the spacing as needed
X_subsampled = X(1:arrow_spacing:end, 1:arrow_spacing:end);
Y_subsampled = Y(1:arrow_spacing:end, 1:arrow_spacing:end);
Ex_subsampled = Ex_norm(1:arrow_spacing:end, 1:arrow_spacing:end);
Ey_subsampled = Ey_norm(1:arrow_spacing:end, 1:arrow_spacing:end);

% Create a figure for streamlines with arrowheads
figure;
quiver(X_subsampled, Y_subsampled, Ex_subsampled, Ey_subsampled, 'b');
title('Distribution of equipotential lines and steamlines with normalized arrowheads - Case 1 - 12110623 曹正阳');
xlabel('X-axis (m)');
ylabel('Y-axis (m)');

% Add equipotential lines on top of potential plot
hold on;
contour(X, Y, V, 1 , 'k', 'LineWidth', 1);
hold off;

C. Case 3: Electric Field Distribution of three identical point charges located at the vertices of an equilateral triangle;

%% CASE3
% Constants and parameters
k = 8.99e9;  % Electrostatic constant
Q1 = 8e-9;   % Charge 1 (in Coulombs)
Q2 = Q1;   % Charge 2 (in Coulombs)
Q3 = Q2;   % Charge 3 (in Coulombs)

% Define the range of the field
xm = 4;  % Adjust the range as needed
ym = 4;

x = linspace(-xm, xm, 800);
y = linspace(-ym, ym, 800);

[X, Y] = meshgrid(x, y);

% Calculate the distances from each charge to all points in the field
R1 = sqrt((X + sqrt(3)).^2 + (Y + 1).^2);
R2 = sqrt((X - sqrt(3)).^2 + (Y + 1).^2);
R3 = sqrt(X.^2 + (Y - 2).^2);

% Calculate the electric potentials due to each charge
V1 = k * Q1 ./ R1;
V2 = k * Q2 ./ R2;
V3 = k * Q3 ./ R3;

% Calculate the total electric potential
V = V1 + V2 + V3;

% Calculate the electric field components
[Ex, Ey] = gradient(-V);

figure;
mesh(X, Y, V);
